layout(points) in;
layout(triangle_strip, max_vertices=4) out;

out vec2 g_texcoord;

void emitPoint(vec4 center, float posX, float posY, float texX, float texY)
{
	vec4 p = center + vec4(posX, posY, 0, 0);
	vec2 t = vec2(texX, texY);
	gl_Position = p * projMatrix;
	g_texcoord = t;
	EmitVertex();
}

void main()
{
	vec4 center = gl_in[0].gl_Position;
	center *= viewMatrix;
	emitPoint(center, -halfBBX, halfBBY, 0, 1);
	emitPoint(center, -halfBBX, -halfBBY, 0, 0);
	emitPoint(center, halfBBX, -halfBBY, 1, 0);
	emitPoint(center, halfBBX, halfBBY, 1, 1);
}
